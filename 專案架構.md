```
glux-compiler/
├── Cargo.toml                // Cargo 設定檔，管理 Rust 依賴與編譯參數
├── src/                      
│   ├── main.rs               // 專案入口，負責初始化並串連各個模組
│   ├── lexer.rs              // 詞法分析，將原始程式碼分割成 tokens
│   ├── parser.rs             // 語法解析，生成抽象語法樹（AST），遇到關鍵字則呼叫對應處理器
│   ├── ast.rs                // 定義各種 AST 節點結構（例如 if、while、函式呼叫等）
│   ├── semantic.rs           // 語義分析與型別檢查，負責符號表與作用域管理
│   ├── codegen.rs            // LLVM IR 生成，利用 Inkwell 將 AST 轉成 LLVM 指令
│   ├── error.rs              // 集中處理所有錯誤訊息
│   ├── builtin/              // 內建函數模組（例如 print、len 等）
│   │   ├── mod.rs            // 宣告內建函數相關子模組
│   │   └── builtin.rs        // 定義並處理內建函數的邏輯
│   └── keyword/              // 關鍵字處理器模組，讓每個關鍵字的解析獨立實作
│       ├── mod.rs            // 宣告 keyword 模組下的所有子模組
│       ├── keyword.rs        // 定義關鍵字處理的 trait 與通用接口
│       ├── if_handler.rs     // 專門解析 if 語句的處理器
│       ├── while_handler.rs  // 專門解析 while 語句的處理器
│       ├── for_handler.rs    // 專門解析 for 語句的處理器（根據需要擴充）
│       └── keyword_registry.rs // 註冊所有關鍵字與其對應處理器的表
└── stdlib/                   // 標準函式庫（用 Glux 自己撰寫，便於多語言移植）
    ├── stdlib.glux         // 標準函式庫主入口，整合各子模組
    ├── math.glux           // 數學相關函式與常數
    ├── io.glux             // 輸入/輸出相關函式
    ├── string.glux         // 字串處理相關函式
    └── ...                 // 可依需求擴充其他子函式庫模組
```
